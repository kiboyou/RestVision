    @FXML
    void Add(ActionEvent event) {


        String produit_nom,description,prix,quantite,fournisseur_nom,adresse,ville,telephone;
        produit_nom = txtProduit_Nom.getText();
        description = txtDescription.getText();
        prix = txtPrix.getText();
        quantite = txtQuantite.getText();
        fournisseur_nom = txtFournisseur.getText();
        adresse = txtAdresse.getText();
        ville = txtVille.getText();
        telephone = txtTelephone.getText();
        try
        {
            pst = con.prepareStatement("insert into stock(produit_nom,description,prix,quantite,fournisseur_nom,adresse,ville,telephone)values(?,?,?,?,?,?,?,?)");
            pst.setString(1, produit_nom);
            pst.setString(2, description);
            pst.setString(3, prix);
            pst.setString(4, quantite);
            pst.setString(5, fournisseur_nom);
            pst.setString(6, adresse);
            pst.setString(7, ville);
            pst.setString(8, telephone);

            pst.executeUpdate();

            Alert alert = new Alert(Alert.AlertType.INFORMATION);
            alert.setTitle("GESTION DU STOCK");

            alert.setHeaderText("Student Registation");
            alert.setContentText("Record Addedddd!");
            alert.showAndWait();
            table();

            txtProduit_Nom.setText("");
            txtDescription.setText("");
            txtPrix.setText("");
            txtQuantite.setText("");
            txtFournisseur.setText("");
            txtAdresse.setText("");
            txtVille.setText("");
            txtQuantite.setText("");
            txtProduit_Nom.requestFocus();
        }
        catch (SQLException ex)
        {
            Logger.getLogger(FXMLDocumentController.class.getName()).log(Level.SEVERE, null, ex);
        }
